Specific atom struct functions

  1. add_H_atom(atom,Box_dim,ind) % This function protonates one or two sites in the atom struct
  2. add2atom(XYZ_labels,XYZ_data,varargin) % Append XYZ atomtype labels and XYZ data to an existing atom struct.
  3. adjust_H_atom(atom,Box_dim) % Adjust hydrogen atoms in the atom struct.
  4. cat_atom(atom_1,atom_2) % Concatenate two atom structs.
  5. closest_atom(atom,Box_dim,ref_atom) % Find the closest atom to the reference atom.
  6. copy_atom(atom,atomtype,new_atomtype,new_resname,trans_vec,varargin) % Copy and translate atoms in the atom struct.
  7. create_atom(type,resname,limits,nmax,varargin) % Create new atoms, useful for adding ions to a system.
  8. create_grid_atom(atom_label,nM,limits,dim,varargin) % Put ions on a grid plane and add them to an atom struct.
  9. duplicate_atom(atom,molID) % Duplicate residue with molid MolID.
  10. fuse_atom(atom,Box_dim,varargin) % Fuse all sites within a certain cutoff distance.
  11. heal_atom(atom,Box_dim,ind,varargin) % Heal sites in the atom struct by adding a certain atom type.
  12. ionize_atom(type,resname,limits,nmax,varargin) % Add ions within a certain region defined by limits.
  13. insert_atom(atom,new_atom,position) % Insert a new atom at the specified position.
  14. merge_atom(atom1,Box1,atom2,type,Atom_label,r) % Merge atom structs based on distance criteria.
  15. molid_rotate(atom,Box_dim,MolID,rotate_dim) % Rotate the atom struct based on MolID.
  16. molid_translate(atom,trans_vec,MolID) % Translate a specific molecule ID.
  17. noupdate_atom(atom) % Prevent updating of certain properties in the atom struct.
  18. occupancy_atom(atom,Box_dim) % Calculate occupancy of atoms within the box dimensions.
  19. overwrite_atom(In_atom,atomtype,resname) % Overwrite atom struct information with new data.
  20. place_atom(atom,position) % Place the atom struct at the specified position.
  21. position_molid(atom,position_vec,MolID) % Move a molecule ID to a certain position.
  22. protonate_atom(atom,Box_dim,varargin) % Protonate specified sites in the atom struct.
  23. remove_molid(atom,MolID) % Remove residue with a specific molecule ID.
  24. remove_occypancy_atom(atom) % Remove particles with identical coordinates to preceding ones.
  25. remove_residues(atom,resnames,lo,hi,dim) % Remove residues between specified limits in the simulation box.
  26. remove_resname(atom,resnames) % Remove residues with specified names.
  27. remove_SOL(atom,atomname,lo,hi,dim) % Remove solvent residues between specified limits.
  28. remove_type(atom,typescell) % Remove atom types specified in typescell.
  29. rename_atom(atom,old_name,new_name) % Rename atoms in the atom struct.
  30. rename_type(atom,atomtype,new_atomtype,varargin) % Rename atom types in the atom struct.
  31. replicate_atom(atom,Box_dim,replicate) % Replicate the atom struct along orthogonal dimensions.
  32. replace_atom(new_atom,prev_atom,molid_index) % Replace molecule ID in an atom struct with a new atom struct.
  33. resname_atom(atom) % Guess residue names for all atom types.
  34. slice_atom(atom,limits,invert) % Slice the atom struct within specified limits.
  35. slice_box(atom,Box_dim,limits) % Slice a simulation box within given limits.
  36. slice_molid(atom,limits,invert) % Slice molecules within specified limits.
  37. slice_triclinic_atom(atom,limits,invert) % Slice a triclinic atom struct within limits.
  38. solvate_atom(limits,density,r,maxsol,solute_atom,varargin) % Generate a solvent structure within specified limits.
  39. spc2tip4p(atom) % Convert SPC water molecules to TIP4P model.
  40. spc2tip5p(atom) % Convert SPC water molecules to TIP5P model.
  41. spce2tip4p(atom) % Convert SPC/E water molecules to TIP4P model.
  42. sphere_atom(atom,Box_dim,center,radius) % Create a spherical region of atoms.
  43. substitute_atom(atom,Box_dim,NumOctSubst,O1,O2,minO2O2_dist,varargin) % Perform isomorphous substitution of atoms.
  44. substitute_NonCentroSymm_atom(atom,Box_dim,replace_type,varargin) % Substitute non-centrosymmetric atoms.
  45. tile_atom(atom,scale_vec,Box_dim,Resname) % Tile the atom struct in a specific direction.
  46. tip3p2tip4p(atom) % Convert TIP3P water molecules to TIP4P model.
  47. translate_atom(atom,trans_vec,Resname) % Translate a residue by a specified vector.
  48. translate_molid(atom,trans_vec,molid) % Translate a molecule ID by a specified vector.
  49. tube_atom(atom,scale_vec,Box_dim,Resname) % Create a nanotube structure from the atom struct.
  50. update_atom(atom) % Update molecule and atom indices in the atom struct.
