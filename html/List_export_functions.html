Export of structure/topology files

  1. write_atom_all(atom,Box_dim,filename_out,varargin) % Write various file types for the atom struct, best suited for Clayff systems.
  2. write_atom_cif(atom,Box_dim,filename_out) % Write a basic .cif file from the atom struct.
  3. write_atom_gro(atom,Box_dim,filename_out) % Write a .gro file from the atom struct, optionally including velocities.
  4. write_atom_itp(atom,Box_dim,filename_out,varargin) % Create and print a Gromacs .itp file, best for Clayff or interface force fields.
  5. write_atom_lmp(atom,Box_dim,filename_out,varargin) % Create and print a LAMMPS data file (.lj), suited for Clayff systems.
  6. write_atom_mol2(atom,Bond_index,Box_dim,filename_out) % Write a .mol2 file from the atom struct.
  7. write_atom_oplsaa_go_itp(atom,Box_dim,filename_out,varargin) % Create and print a Gromacs .itp file for OPLS-AA or GO systems.
  8. write_atom_pdb(atom,Box_dim,filename_out) % Write a .pdb file from the atom struct using Gromacs.
  9. write_atom_pqr(atom,Box_dim,filename_out,varargin) % Write a .pqr file from the atom struct.
  10. write_atom_psf(atom,Box_dim,filename_out,varargin) % Write a .psf file from the atom struct.
  11. write_atom_xyz(atom,Box_dim,filename_out) % Write an XYZ file from the atom struct.
  12. write_atom(atom,Box_dim,filename_out,varargin) % Write different file types (.gro, .pdb, .xyz, .itp, etc.) based on filename and parameters.
  13. write_atom_sdf(atom,Box_dim,filename_out) % Write an SDF file from the atom struct.
  14. write_atom_dodecahedron_gro(atom,Box_dim,filename_out) % Write a .gro file using a dodecahedron-shaped simulation box.
  15. write_atom_top(atom,Box_dim,filename_out) % Write a topology file (.top) from the atom struct.
  16. write_itp(itp,filename_out) % Write a molecular topology file (.itp) from an itp struct.
  17. CONECT_atom(atom,Box_dim,filename_out) % Write CONECT records for a PDB file.

Export of trajectory files

  1. write_gro_traj(atom,traj,Box_dim,filename_out) % Write a .gro trajectory file.
  2. write_atom_multiple_gro(atom,traj,filename_out) % Write multiple .gro files for trajectory output.
  3. write_pdb_traj(atom,traj,Box_dim,filename_out) % Write a .pdb trajectory file.
  4. write_xyz_traj(atom,traj,Box_dim,filename_out) % Write a .xyz trajectory file.
  5. write_ave_gro(atom,traj,Box_dim,filename_out) % Write an average structure from a .gro trajectory.
  6. write_ave_pdb(atom,traj,Box_dim,filename_out) % Write an average structure from a .pdb trajectory.

Export of other formats and data

  1. export_ndx(atom,Box_dim,filename_out) % Export an index file (.ndx) from the atom struct.
  2. write_xvg(filename, data) % Export data in Gromacs .xvg format.
  3. write_tabulated_potentials(filename, data) % Write tabulated potential files.
  4. write_ff(atom,filename_out) % Write force field parameters.
  5. write_ffnonbonded(atom,filename_out) % Write non-bonded force field parameters.
  6. write_ffnonbonded_C6C12(atom,filename_out) % Write non-bonded parameters (C6, C12) for a force field.

Miscellaneous

  1. replace_string(filename_in,filename_out,old_string,new_string) % Replace strings in files.
