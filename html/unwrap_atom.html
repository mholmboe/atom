        xshift=0;yshift=0;zshift=0;
        if [atom(1).med_x]>Box_dim(1)
            xshift=-Box_dim(1);
        elseif [atom(1).med_x]<0
            xshift=Box_dim(1);
        end

            zshift=-Box_dim(3);
        elseif [atom(1).med_z]<0
            zshift=Box_dim(3);
        end

        atom=translate_atom(atom,[xshift yshift zshift],'all');
