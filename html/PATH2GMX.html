  1. PATH2GMX % Can be used to invoke Gromacs utilities, see for instance the gmx() function
  2. PATH2GMX('add') % adds Gromacs to the MATLAB path
function PATH2GMX = PATH2GMX(varargin)

PATH2GMX ='/usr/local/gromacs-2023.3/bin'; % Note, this is a version compiled wth mpi

if nargin>0

    PATH=getenv('PATH');
    if regexp(PATH,strcat(':',PATH2GMX))
        PATH=strrep(PATH,strcat(':',PATH2GMX),'');
        setenv('PATH',PATH);
    end
    setenv('PATH', [getenv('PATH'),':',PATH2GMX]);
    % Now you could try
    % system('gmx editconf -h')

end

end
