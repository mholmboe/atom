Convert water functions

  1. spc2tip4p(filename) % Convert a .gro or .pdb file with spc water to tip4p water.
  2. spc2tip5p(filename) % Convert a .gro or .pdb file with spc water to tip5p water.
  3. spce2tip4p(filename) % Convert a .gro or .pdb file with spce water to tip4p water.
  4. tip3p2tip4p(filename) % Convert a .gro file with tip3p water to tip4p water.

Various distance and bond functions

  1. atom_stats(atom,Box_dim) % Generate statistics about atom types, coordination, and charges in the structure.
  2. bond_angle_atom(atom,Box_dim,max_short_dist,max_long_dist,varargin) % Find all bonds and angles of the atom struct. 'More' is an optional argument.
  3. bond_angle_dihedral_atom(atom,Box_dim,varargin) % Find all bonds, angles, and dihedrals of the atom struct. Optional arguments include Rmaxshort and Rmaxlong.
  4. bond_angle_type(atom1,atom2,Box_dim,rmin,rmax,angle_limit,varargin) % Find all bonds and angles of the atom types.
  5. bond_valence_atom(atom,Box_dim,varargin) % Calculate bond valence values using the bond valence method.
  6. bond_valence_data(ion1,ion2,R,varargin) % Fetch data to calculate bond valence values for specified atom types.
  7. dist_matrix_atom(atom,Box_dim) % Calculate the distance matrix from the atom struct.
  8. neigh_atom(atom,Box_dim,rmax,varargin) % Check neighbors for each atom and output their information.
  9. neighbor_func(solute_index,XYZ_solute,XYZ_data,Box_dim,radius) % Scan xyz data and check neighbors within a specified radius.
  10. radius_atom(atom,ffname,watermodel) % Fetch the ion radius from Clayff, Interface, or Interface2015 force fields.
  11. radius_ion(Atom_label) % Fetch the ionic radius.
  12. radius_vdw(Atom_label) % Fetch the van der Waals radius.
  13. rdf_atom(atom,Box_dim,varargin) % Calculate the radial distribution function and coordination number.
  14. xrd_atom(varargin) % Calculate theoretical XRD patterns from a .pdb, .gro file, or atom struct.

Other general functions

  1. add2atom(XYZ_labels,XYZ_data,varargin) % Append XYZ atom type labels and XYZ data to an existing atom struct.
  2. analyze_atom(atom,Box_dim) % Fetch properties of atoms using methods like bond valence and radii.
  3. atomic_scattering_factors(Atom_label,lambda,twotheta,DW) % Retrieve atomic scattering factors vs. 2theta.
  4. ave_atom(atom) % Calculate the mean of the atom coordinates.
  5. Box_dim2Cell(Box_dim) % Transform the Box_dim variable to the Cell variable.
  6. Cell2Box_dim(Cell) % Transform the Cell variable into the Box_dim variable.
  7. COM_atom(atom,MolID) % Calculate the center of mass (COM) for specified elements.
  8. COM_molid(atom,MolID) % Calculate the COM for specific elements.
  9. COM_SOL(MolID,XYZ_data,Atom_label,XYZ_labels,Box_dim) % Compute the COM of SPC water.
  10. composition_atom(atom) % Analyze the composition of the atom struct.
  11. density_atom(atom,Box_dim) % Calculate concentration, electron density profiles, and charge density.
  12. density_atom_fast(atom,Box_dim) % A faster version of the density_atom function.
  13. draw_box_atom(Box_dim,LineColor,LineThickness) % Draw a box to visualize dimensions.
  14. element_atom(atom,varargin) % Convert atom type names to element names.
  15. element_color(Atom_label) % Assign a specific color to each element.
  16. frac2atom(atom,Box_dim,angleparam,angletype) % Transform fractional coordinates to Cartesian coordinates.
  17. frame2atom(atom,traj,frame,Box_dim,varargin) % Extract a frame to the trajectory matrix.
  18. G2_atom(atom,Box_dim) % Calculate the continuous G2 factor.
  19. hist_atom(atom,s) % Calculate density profiles in the X, Y, or Z direction.
  20. histz_atom(atom,s) % Calculate density profiles in the Z direction.
  21. molecule_atom(atom,varargin) % Set molecule ID, residue name, and element names of the atom struct.
  22. median_atom(atom) % Calculate the median position of the atom struct.
  23. neutralize_atom(atom) % Set the charge of all atom types to zero.
  24. orto_atom(atom,Box_dim) % Transform a triclinic atom struct to an orthogonal one.
  25. place_atom(atom,position) % Place the atom struct at a specified position.
  26. plot_density_atom(atom,Box_dim,varargin) % Draw the atom struct in 3D along with density profiles.
  27. plot_atom(atom,Box_dim,varargin) % Draw the atom struct in 3D.
  28. plot_xvg(filename) % Plot .xvg files, typically used in GROMACS simulations.
  29. properties_atom(atom) % Analyze properties of the atom struct.
  30. reduced_mass(Atom_label1,varargin) % Calculate the reduced mass.
  31. show_atom(atom,varargin) % Draw the atom struct in 3D with additional features.
  32. show_arrow(p1,p2,varargin) % Plot a 3D arrow as a patch object.
  33. show_axis(varargin) % Draw the axis in a plot.
  34. show_box(Box_dim) % Draw the simulation box.
  35. show_Hbonds_atom(atom) % Display or calculate hydrogen bonds in the atom struct.
  36. show_miller(Box_dim) % Draw the Miller planes of the Box_dim/Cell variables.
  37. show_density_atom(atom) % Display density of atoms.
  38. triclinic_atom(atom,Box_dim,angleparam,angletype) % Transform an orthogonal atom struct to a triclinic one.
  39. update_atom(atom) % Update molecule and atom indices in the atom struct.
  40. vmd(atom,Box_dim) % Plot the atom struct using VMD.
  41. closest_atom(atom,Box_dim,ref_atom) % Find the closest atom to the reference atom.
  42. insert_atom(atom,new_atom,position) % Insert a new atom at the specified position.
  43. remove_duplicate_atom(atom) % Remove duplicate atoms from the atom struct.
  44. sort_atom(atom,Box_dim,varargin) % Sort the atom struct based on specified properties.
  45. split_atom(atom,Box_dim,varargin) % Split the atom struct into separate parts based on criteria.
  46. translate_atom(atom,trans_vec) % Translate the atom struct by a specified vector.
  47. rotate_atom(atom,rotation_matrix) % Rotate the atom struct by a specified matrix.
  48. spiral_atom(atom,Box_dim,varargin) % Create or modify spiral structures in the atom struct.
  49. round_atom(atom) % Round atom positions or coordinates.
  50. mass_atom(atom) % Calculate or define mass for atoms.
  51. radius_crystal(Atom_label) % Fetch or calculate ionic radii for crystal structures.
  52. sigma_vdw(Atom_label) % Compute sigma values for van der Waals interactions.
  53. Bragg(varargin) % Calculate Bragg peaks for crystallographic or XRD data.
  54. unreplicate_atom(atom) % Remove replicated atoms from the structure.

Keep/remove functions

  1. keep_atom(atom,resname) % Keep only specified residue names in the atom struct.
  2. keep_resname(atom,resnames) % Keep only specified residue names.
  3. remove_molid(atom,MolID) % Remove residues with specified molecule IDs.
  4. remove_occypancy_atom(atom) % Remove particles with identical coordinates to preceding ones.
  5. remove_residues(atom,resnames,lo,hi,dim) % Remove residues within specified limits.
  6. remove_resname(atom,resnames) % Remove residues by name.
  7. remove_SOL(atom,atomname,lo,hi,dim) % Remove solvent residues within specified limits.
  8. remove_type(atom,typescell) % Remove specified atom types.
